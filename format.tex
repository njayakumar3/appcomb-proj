%% Links
\def\link#1{%
  \pdfstartlink
    user{/Subtype /Link
    /Border [ 0 0 0 ]
    /A <<
      /Type /Action
      /S /URI
      /URI (#1)
    >>}%
    {\it #1}%
  \pdfendlink
}
